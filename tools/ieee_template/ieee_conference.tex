% ----------------------------------------------------------------------
% Column-overlap mitigations (added 2026-04-27 after appendix overflow
% in paper_draft_ieee_v2.pdf). IEEE 2-column at 10pt has a ~3.4-inch
% column width; without these, long unbreakable tokens (file paths,
% git SHAs, regex patterns, URLs) overrun into the next column.
% ----------------------------------------------------------------------

% Allow LaTeX more flexibility in inter-word spacing so paragraphs
% always break SOMEWHERE rather than overrunning. Combined with
% \emergencystretch this prevents the "extra column-bleed" cases.
\sloppy
\setlength{\emergencystretch}{3em}

% Allow hyphenation inside \texttt{...} so file paths, identifiers,
% and code-style spans break at slash/underscore/dot/dash. fontspec
% would do this automatically but newunicodechar interactions make
% an explicit setup safer.
\usepackage[htt]{hyphenat}

% URLs: let url package break at any character if needed, not just
% at conventional break points (slashes, dots).
\usepackage{xurl}

% Verbatim code blocks: with fvextra loaded, we can globally set breaklines
% on the Verbatim environment. The build_ieee_pdf.py post-processor rewrites
% pandoc's plain `verbatim' env to fvextra's `Verbatim' so these settings
% apply to every fenced code block in the markdown source.
\fvset{breaklines=true,breakanywhere=true,fontsize=\small,
       breaksymbolleft={\tiny\ensuremath{\hookrightarrow}}}

% Reduce penalty for ending a column with an underfull box, which is
% kinder to layout than letting boxes overrun.
\hbadness=10000
\vbadness=10000

% ----------------------------------------------------------------------
% Section heading style override.
%
% IEEEtran's conference mode renders \section in centered small caps
% (\scshape), \subsection in italics (\itshape), and \subsubsection in
% italics. We render in BOLD instead so headings have stronger visual
% weight (the build pipeline post-processes \section -> \section* to
% suppress IEEE's auto Roman/letter prefix because the markdown source
% carries explicit numbers like '5. Structural Remediation' that the
% body's 44+ section cross-references target by number).
%
% Definitions are copied from IEEEtran.cls (conference branch, ~line
% 5467) with the trailing format command swapped (\scshape/\itshape
% -> \bfseries / \bfseries\itshape).
% ----------------------------------------------------------------------
\makeatletter
\def\section{\@startsection{section}{1}{\z@}%
  {1.5ex plus 1.5ex minus 0.5ex}%
  {0.7ex plus 1ex minus 0ex}%
  {\normalfont\normalsize\centering\bfseries}}
\def\subsection{\@startsection{subsection}{2}{\z@}%
  {1.5ex plus 1.5ex minus 0.5ex}%
  {0.7ex plus .5ex minus 0ex}%
  {\normalfont\normalsize\bfseries}}
\def\subsubsection{\@startsection{subsubsection}{3}{\parindent}%
  {0ex plus 0.1ex minus 0.1ex}%
  {0ex}%
  {\normalfont\normalsize\bfseries\itshape}}
\makeatother
